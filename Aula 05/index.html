      // Função para remover o outline de todos os elementos
      function removeOutline() {
        // Seleciona todos os elementos da página
        const elements = document.querySelectorAll('*');
        
        // Itera sobre todos os elementos e define outline como none
        elements.forEach(element => {
          element.style.outline = 'none';
        });
      }
      
      // Chama a função para remover o outline
      removeOutline();
